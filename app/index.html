WebView 控制面板

JavaScript API 示例


// 最小化窗口
mata.win.minimize()

// 最大化窗口
mata.win.maximize()

// 恢复窗口
mata.win.restore()

// 关闭窗口
mata.win.close()

// 居中窗口
mata.win.center()

// 调整窗口大小 (宽, 高)
mata.win.resize(800, 600)

// 设置窗口标题
mata.win.title('新标题')

// 显示提示框
mata.alert('消息内容')

// 窗口最小化事件
document.addEventListener("mata-win-minimized", function() {
    console.log('窗口已最小化');
});

// 窗口恢复事件
document.addEventListener("mata-win-restored", function() {
    console.log('窗口已恢复');
});

// 窗口最大化事件
document.addEventListener("mata-win-maximized", function() {
    console.log('窗口已最大化');
});
